async function gerarProvaComIA() { const material = document.getElementById('materialIA').value; const qtd = document.getElementById('quantidadeQuestoes').value || 10; if(!material.trim()) { alert('Cole um texto ou faça upload!'); return; } if(!API_KEY) { alert('Configure sua chave API primeiro!'); return; } document.getElementById('loadingIA').style.display = 'block'; const prompt = `Crie ${qtd} perguntas de múltipla escolha com 4 alternativas. Responda APENAS JSON: {"titulo":"Título","perguntas":[{"texto":"Pergunta?","alternativas":["A","B","C","D"],"correta":0}]} Material: ${material.substring(0, 3000)}`; // CORREÇÃO: Usar gemini-1.5-flash (modelo atualizado) const model = 'gemini-1.5-flash'; try { const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${API_KEY}`; const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], generationConfig: { temperature: 0.7, maxOutputTokens: 2048 } }) }); const data = await response.json(); document.getElementById('loadingIA').style.display = 'none'; // CORREÇÃO: Verificar erros específicos da API if (!response.ok) { if (data.error) { if (data.error.status === 'NOT_FOUND') { alert('❌ Modelo não encontrado. Possíveis causas:\n\n' + '1. API não ativada no Google Cloud Console\n' + '2. Link para ativar: https://console.cloud.google.com/apis/library/generativelanguage.googleapis.com\n' + '3. Após ativar, aguarde 2-3 minutos\n\n' + 'Erro técnico: ' + data.error.message); } else if (data.error.status === 'PERMISSION_DENIED') { alert('❌ API não autorizada!\n\n' + 'Solução: Acesse o link abaixo e clique em ATIVAR:\n' + 'https://console.cloud.google.com/apis/library/generativelanguage.googleapis.com'); } else if (data.error.message && data.error.message.includes('API key')) { alert('❌ Chave API inválida!\n\n' + 'Gere uma nova chave em:\n' + 'https://aistudio.google.com/app/apikey'); } else { alert('❌ Erro: ' + data.error.message); } } else { alert('❌ Erro desconhecido na API'); } return; } // CORREÇÃO: Verificar se a resposta tem conteúdo if(!data.candidates || !data.candidates[0] || !data.candidates[0].content) { alert('❌ Resposta vazia da IA. Tente novamente.'); return; } let texto = data.candidates[0].content.parts[0].text; texto = texto.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim(); const prova = JSON.parse(texto); if(!prova.perguntas || prova.perguntas.length === 0) { alert('❌ Nenhuma pergunta gerada.'); return; } preencherFormulario(prova); } catch(error) { document.getElementById('loadingIA').style.display = 'none'; console.error('Erro completo:', error); alert('❌ Erro: ' + error.message + '\n\n' + 'Verifique se:\n' + '1. A API está ATIVADA em: https://console.cloud.google.com/apis/library/generativelanguage.googleapis.com\n' + '2. Sua chave API é válida\n' + '3. Você tem conexão com a internet'); } }
