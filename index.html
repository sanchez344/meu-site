async function gerarProvaComIA() { const material = document.getElementById('materialIA').value; const qtd = document.getElementById('quantidadeQuestoes').value || 10; if (!material.trim()) { alert('Cole um texto ou faça upload!'); return; } if (!API_KEY) { alert('Configure sua chave API Gemini primeiro!'); return; } document.getElementById('loadingIA').style.display = 'block'; const prompt = ` Crie ${qtd} perguntas de múltipla escolha com 4 alternativas. Responda SOMENTE em JSON válido neste formato: { "titulo": "Título da prova", "perguntas": [ { "texto": "Pergunta?", "alternativas": ["A", "B", "C", "D"], "correta": 0 } ] } Material: ${material.substring(0, 6000)} `; try { // MODELO ATUAL FUNCIONANDO const model = "gemini-1.5-flash"; const response = await fetch( `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${API_KEY}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [ { parts: [ { text: prompt } ] } ], generationConfig: { temperature: 0.7, maxOutputTokens: 4096 } }) } ); if (!response.ok) { const erroTexto = await response.text(); throw new Error(`Erro HTTP ${response.status}: ${erroTexto}`); } const data = await response.json(); console.log(data); if (!data.candidates || !data.candidates.length) { throw new Error("A API não retornou respostas."); } let texto = data.candidates[0].content.parts[0].text; // limpa markdown texto = texto .replace(/```json/g, '') .replace(/```/g, '') .trim(); const prova = JSON.parse(texto); if (!prova.perguntas || !prova.perguntas.length) { throw new Error("JSON inválido retornado pela IA."); } preencherFormulario(prova); } catch (e) { console.error(e); alert( "Erro ao comunicar com Gemini:\n\n" + e.message ); } finally { document.getElementById('loadingIA').style.display = 'none'; } }
